    & "{impersonationLevel=impersonate}!\\" & strComputer & "\root\cimv2")

    ("Win32_Printer.DeviceID='ArtDepartmentPrinter'")

objPrinter.PortName = "IP_192.168.1.10"
objPrinter.Put_
