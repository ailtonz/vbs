Dim fso, d, dc, s, n, WshShell
  Set fso = CreateObject("Scripting.FileSystemObject")
  Set dc = fso.Drives

do until foundCD = true
	  For Each d in dc
		 If d.IsReady Then
		setuppath = d.DriveLetter & ":\clients\win9x\dsclient.exe"
		if fso.FileExists(setuppath) then
		   source = d.DriveLetter & ":\*.*"
		   dest = " C:\winnt4s\"
		   args = "/E /H /Y"
			  set WshShell = CreateObject("WScript.Shell")
			  foundCD = True
			  fso.CreateFolder "C:\winnt4s\"
			  WshShell.Run "xcopy " & source & dest & args, 0, True
		   Exit For
		end if
		end if
	  Next
	  if foundCD = false then 
		msgbox "Please Insert the Microsoft Windows NT 4 Server compact disc into any cdrom drive.", VBOkOnly + VBSystemModal ,"Automated Classroom Setup"
	  end if
loop
